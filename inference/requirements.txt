# DARWIN inference layer dependencies.

# IMPORTANT: install with python3.11 (/opt/homebrew/bin/python3.11).
# Python 3.14 lacks prebuilt wheels for mlx and its native deps; 3.11 is the
# supported interpreter for everything MLX-related on this machine.
#
#   /opt/homebrew/bin/python3.11 -m venv .venv
#   .venv/bin/pip install -r inference/requirements.txt
#
# MLX runs on the Apple GPU via Metal (Apple Silicon only).
#
# SUPPLY-CHAIN: direct deps are pinned to EXACT versions (`==`), not floating
# `>=` lower bounds. A `>=` range silently pulls whatever the newest release is at
# install time — so a single compromised or yanked PyPI upload lands in the
# inference process (which loads models and runs on-device) with no review. These
# pins are the known-good versions from the verified python3.11 venv. To upgrade
# deliberately: bump a pin, reinstall, test, commit. To refresh from a working
# venv: `.venv/bin/pip freeze | grep -iE '<the names below>'`.

coremltools==9.0  # Core ML dispatch. ALSO the runtime for the DEFAULT op=embed backend (inference/coreml_embed.py: BAAI/bge-small-en-v1.5 -> Core ML, ANE-eligible) + Phase-3 aux models (scripts/ane_probe.py)

# op=embed CORE ML SENTENCE EMBEDDER (inference/coreml_embed.py) — the DEFAULT
# op=embed backend. transformers supplies the tokenizer at RUNTIME (fast /
# tokenizers-backed for bge's WordPiece — no torch needed to tokenize) and drives
# the ONE-TIME model conversion. torch is needed ONLY for that one-time
# convert-on-first-use (jit.trace -> coremltools) + the faithfulness reference;
# steady-state prediction after conversion uses ONLY coremltools + the tokenizer.
# HEAVY: torch is a large wheel (~hundreds of MB, the heaviest dep here) — the
# house otherwise avoids it (see the misaki note below). IMPORT-GUARDED: server.py
# and coreml_embed.py import fine (py_compile / pyflakes pass) WITHOUT these; if
# they are absent the Core ML embedder cannot build and op=embed HONESTLY falls
# back to the resident-4B mean-pool path (embedder="llm-qwen3-4b-meanpool",
# fell_back=true) instead of breaking. Pinned to the known-good conversion combo.
transformers==5.11.0
torch==2.12.0

# LEARNED VAD (inference/coreml_vad.py) — the Core ML Silero VAD backend that
# replaces the RMS energy gate (measured win under inference/benchmarks/vad_eval/).
# silero-vad ships the bundled ~1.3MB v5 checkpoint (silero_vad.jit) the backend
# converts to Core ML ONE-TIME via the torch above (jit.trace -> coremltools);
# steady-state prediction uses ONLY coremltools. soundfile reads the test-set WAVs
# in the eval harness. IMPORT-GUARDED like the embedder: coreml_vad.py imports fine
# (py_compile / pyflakes pass) WITHOUT these; if absent the learned VAD cannot build
# and the daemon HONESTLY falls back to the RMS gate (surfaced, never silent).
silero-vad==6.2.1
soundfile==0.14.0
